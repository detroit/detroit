    def release
      pkgs = Pathname.new(pkgdir).glob("*-#{version}.gem")
      if pkgs.empty?
        report "No .gem packages found for version {version} at #{pkgdir}."
      else
        pkgs.each do |file|
          sh "gem push #{file}"
        end
      end
    end
