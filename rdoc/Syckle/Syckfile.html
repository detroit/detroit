    def method_missing(sym, *args, &block)
      sym = sym.to_s.downcase

      if String === args.first
        key = args.first.to_s
        srv = sym.to_s
      else
        key = sym.to_s
        srv = false
      end

      raise SyntaxError if /[=?!]$/ =~ key

      if block_given?
        @hash[key] = Entry.new(&block).to_h
      else
        @hash[key] = {}
      end

      @hash[key]['service'] = srv if srv
      @hash[key]['active']  = false if FalseClass === args.last

      @hash[key]
    end
