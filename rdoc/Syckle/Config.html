    def initialize
      @automatic = false
      @auto_omit = []

      file = Dir['.config/syckle.{yml,yaml}'].first
      if file
        conf = YAML.load(File.new(file))
        conf.each do |k,v|
          __send__("#{k}=", v) if respond_to?("#{k}=")
        end
      end
    end
