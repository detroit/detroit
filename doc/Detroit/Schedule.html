def method_missing(sym, *args, &block)
  service_class = sym.to_s
  case service_class
  when /^[A-Z]/
    if Hash === args.last
      args.last[:service] = service_class
    else
      args << {:services=>service_class}
    end
    case args.first
    when String, Symbol
      name = args.first
    else
      name = service_class.to_s.downcase
    end
    service(name, *args, &block)
  else
    super(sym, *args, &block)
  end
end
