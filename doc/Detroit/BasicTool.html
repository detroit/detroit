def self.assembly(assembly=nil)
  @assembly ||= []
  if assembly
    @assembly << assembly.to_sym
    @assembly.uniq!
  end
  @assembly
end

def self.options(service_class=self)
  service_class.instance_methods.
    select{ |m| m.to_s =~ /\w+=$/ && !%w{taguri=}.include?(m.to_s) }.
    map{ |m| m.to_s.chomp('=') }
end
